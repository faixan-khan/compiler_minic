int main(){
int x,y,z;
int count;
count=0;
for(z=1;z<=100000000;z=z+1){
    for(y=1;y<z;y=y+1){
        for(x=1;x<y;x=x+1){
            x_c=x*x;
            y_c=y*y;
            z_c=z*z;
            if((x_c+y_c)==z_c){
            count=(count+2);
            }
        }
    }
}
print(count);
return 0;
}
